\begin{figure}[tb]
\centering
\caption{\textbf{Canonical truth regimes as prompt-level contracts.}
Each regime specifies an oracle \(O\), stylistic latitude \(\sigma\), and invention license \(\kappa\).
The same surface divergence can be acceptable or hallucinatory depending on whether the prompt binds claims to provided evidence, world facts, or instead licenses invention under explicit constraints.}
\label{fig:canonical-truth-regimes}

\begin{tcolorbox}[
  truthregimecard,
  title={Strict grounded generation / summarization},
  colback=TRBlue,
  colframe=TRBlueFrame
]
\begin{tabularx}{\linewidth}{@{}p{3.4cm}X@{}}
\field{Permissions} &
\sigmakappa{low}{low}. Style is constrained, and new truth-conditional content is not licensed. \\[4pt]

\field{Oracle \(O\)} &
Provided sources, such as documents, tables, or retrieved passages. \\[4pt]

\field{Allowed divergence} &
Paraphrase, compression, and re-ordering; \emph{no new facts}. \\[4pt]

\field{TR violation} &
Any unsupported detail, added entity or number, or contradiction with the provided sources.
\end{tabularx}
\end{tcolorbox}

\begin{tcolorbox}[
  truthregimecard,
  title={Factual task with creative style},
  colback=TRGreen,
  colframe=TRGreenFrame
]
\begin{tabularx}{\linewidth}{@{}p{3.4cm}X@{}}
\field{Permissions} &
\sigmakappa{high}{low}. Style may vary freely, but factual invention remains disallowed. \\[4pt]

\field{Oracle \(O\)} &
World knowledge and/or cited sources. \\[4pt]

\field{Allowed divergence} &
Metaphor, voice, rhythm, vividness, and accessible framing, provided that the propositions remain unchanged. \\[4pt]

\field{TR violation} &
Any unsupported new truth-conditional claim, or any claim that is factually incorrect.
\end{tabularx}
\end{tcolorbox}

\begin{tcolorbox}[
  truthregimecard,
  title={Ideation / fiction-writing},
  colback=TRPurple,
  colframe=TRPurpleFrame
]
\begin{tabularx}{\linewidth}{@{}p{3.4cm}X@{}}
\field{Permissions} &
\sigmakappa{high}{high}. Style may vary freely, and invention is licensed under prompt constraints. \\[4pt]

\field{Oracle \(O\)} &
Prompt constraints, including outline, coherence requirements, genre expectations, and user-provided facts. \\[4pt]

\field{Allowed divergence} &
Invent entities, scenes, alternatives, hypotheses, and speculative possibilities, while respecting the prompt contract. \\[4pt]

\field{TR violation} &
Violation of explicit constraints, such as incoherence, contradiction with user-provided facts, or deviation from a required outline.
\end{tabularx}
\end{tcolorbox}
